1. git init # 创建仓库
2. git add filename # 添加一个文件
3. git commit -m "wrote a file"  # 提交一个文件 -m 是一个参数，添加注释说明
4. git statue # 查看仓库状态，是否有改变
5. git log # 查看仓库日志
6. git log --pretty=oneline # 查看时间线
7. git reset hard --HEAD^   #  回归上一个版本
8. git reset --hard <版本号前几位数字>   # 回归到指定的版本号
9. git reflog  	#  记录每一次命令
10.git add
